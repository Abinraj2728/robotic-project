void loop() {
  float temp = dht.readTemperature();
  int motion = digitalRead(PIR_PIN);
  
  if(temp >= TEMP_THRESHOLD) {
    digitalWrite(RELAY_FAN, HIGH);
  }
  
  if(motion == HIGH && !manualOverride) {
    digitalWrite(RELAY_LAMP, HIGH);
  }
}
